#include <stdio.h>
int main()
{
    int i, j, rows;

    printf("Enter number of rows: ");
    scanf("%d",&rows);
    printf("\n");
    for(i=1; i<=rows; ++i)
    {
        for(j=1; j<=i; ++j)
        {
            printf("%c",64+j);
        }
        printf("\n");
    }
    for(i=rows-1; i>=1; --i)
    {
        for(j=1; j<=i; ++j)
        {
            printf("%c",64+j);
        }
        printf("\n");
    }
    return 0;
}
